namespace = drive_the_narrative

# Hidden event triggered by Drive the narrative decision
# Checks which event to show to the player based on how much supporters they got
drive_the_narrative.0 = {
    hidden = yes


    immediate = {
        every_in_global_list = {
            variable = all_unions
            # story_owner = { save_temporary_scope_as = union_leader }

            if = {
                limit = {
                    story_owner = { is_a_possible_supporter = yes }
                }

                add_to_list = {
                    name = calia_allies
                    value = story_owner
                }
            }
        }

        if = {
            limit = {
                list_size = {
                    name = calia_allies
                    value > 1
                }
            }

            # There are more than one supporter, make the player choose which to keep
            trigger_event = drive_the_narrative.1
        }
        # There are only one supporter, skip to the actual event
        else_if = {
            limit = {
                list_size = {
                    name = calia_allies
                    value = 1
                }
            }
            
            ordered_in_list = {
                list = calia_allies
                position = 0
                save_scope_as = sole_supporter
            }

            set_global_variable = {
                name = chosen_supporter
                value = scope:sole_supporter
            }

            trigger_event = drive_the_narrative.4
        }
        else = {
            debug_log = "[LAST_MENETHIL] Drive the Narrative event triggered with no supporters found."
        }
    }
}

	window = visit_settlement_window
    

    # desc = drive_the_narrative.1.desc

        first_valid = {
            # If the player already saw this screen and went back
            triggered_desc = {

            triggered_desc = {
                trigger = {
                    list_size = {
                        name = calia_allies
                        value > 3
                    }
                }
                desc = drive_the_narrative.1.desc_outro_too_much_supporters
            }
        }
    }

    override_background = {
        reference = council_chamber
    }


    left_portrait = {
        character = root
        animation = thinking
    }


    right_portrait = {
        character = scope:first_ally
        animation = personality_content
    }


    lower_left_portrait = {
        character = scope:second_ally
        animation = personality_content
    }


    lower_center_portrait = {
        character = scope:third_ally
        animation = personality_content
    }
    

    option = {
        name = drive_the_narrative.1.first_ally

        highlight_portrait = scope:first_ally
        
        set_main_supporter = {
            SUPPORTER = scope:first_ally
        }

        set_competition_level = { NB_SUPPORTERS = list_size:calia_allies }

        trigger_event = drive_the_narrative.4
    }
    

    option = {
        name = drive_the_narrative.1.second_ally

        trigger = {
            exists = scope:second_ally
        }

        highlight_portrait = scope:second_ally
        
        set_main_supporter = {
            SUPPORTER = scope:second_ally
        }

        set_competition_level = { NB_SUPPORTERS = list_size:calia_allies }

        trigger_event = drive_the_narrative.4
    }
    

    option = {
        name = drive_the_narrative.1.third_ally

        trigger = {
            exists = scope:third_ally
        }

        highlight_portrait = scope:third_ally
        
        set_main_supporter = {
            SUPPORTER = scope:third_ally
        }

        set_competition_level = { NB_SUPPORTERS = list_size:calia_allies }

        trigger_event = drive_the_narrative.4
    }
    

    # If there are even more allies.
    option = {
        name = drive_the_narrative.1.next_allies

        trigger = {
            list_size = {
                name = calia_allies
                value > 3
            }
        }

        custom_tooltip = {
            text = drive_the_narrative.next_allies_tooltip
        }

        save_scope_value_as = {
            name = seen_first_slide
            value = 1
        }

        trigger_event = drive_the_narrative.2
    }
    

    # If the player doesn't want to continue anymore.
    option = {
        name = drive_the_narrative.abandon

        trigger = {
            list_size = {
                name = calia_allies
                value <= 3
            }
        }

        quit_choosing_my_side_event = yes
    }
    

    immediate = {
        set_variable = {
            name = competition_level
            years = 2
        }

        # This is the most powerful asset Calia has. There should always be at least one
        ordered_in_list = {
            list = calia_allies
            order_by = max_military_strength
            position = 0
            save_scope_as = first_ally
        }

        # If there is a second asset
        if = {
            limit = {
				list_size = {
					name = calia_allies
					value > 1
				}
            }

            ordered_in_list = {
                list = calia_allies
                order_by = max_military_strength
                position = 1
                save_scope_as = second_ally
            }            
        }

        # If there are even a third asset
        if = {
            limit = {
				list_size = {
					name = calia_allies
					value > 2
				}
            }

            ordered_in_list = {
                list = calia_allies
                order_by = max_military_strength
                position = 2
                save_scope_as = third_ally
            }
        }
    }
}

# Drive the narrative, actual event
drive_the_narrative.4 = {
    type = character_event
    title = drive_the_narrative.4.title
    theme = realm
	window = visit_settlement_window
    

    desc = {
        desc = drive_the_narrative.4.desc_intro

        first_valid = {
            # NO REGENCY
            # The supporter is the player's BFF!
            triggered_desc = {
                trigger = {
                    scope:main_supporter = scope:power_holder
                    has_relation_best_friend = scope:main_supporter
                }
                desc = drive_the_narrative.4.desc.bff
            }

            # The supporter is an ally, no need to blackmail them!
            triggered_desc = {
                trigger = {
                    scope:main_supporter = scope:power_holder
                    is_allied_to = scope:main_supporter
                }
                desc = drive_the_narrative.4.desc.allied
            }

            # Yeah clearly here blackmail was necessary
            triggered_desc = {
                trigger = {
                    scope:main_supporter = scope:power_holder
                    has_strong_usable_hook = scope:main_supporter
                }
                desc = drive_the_narrative.4.desc.hooked
            }


            # REGENCY
            # The supporter is the player's BFF!
            triggered_desc = {
                trigger = {
                    scope:main_supporter != scope:power_holder
                    has_relation_best_friend = scope:main_supporter
                }
                desc = drive_the_narrative.4.desc.bff.regent
            }

            # The supporter is an ally, no need to blackmail them!
            triggered_desc = {
                trigger = {
                    scope:main_supporter != scope:power_holder
                    is_allied_to = scope:main_supporter
                }
                desc = drive_the_narrative.4.desc.allied.regent
            }

            # Yeah clearly here blackmail was necessary
            triggered_desc = {
                trigger = {
                    scope:main_supporter != scope:power_holder
                    has_strong_usable_hook = scope:main_supporter
                }
                desc = drive_the_narrative.4.desc.hooked.regent
            }
        }

        desc = drive_the_narrative.4.desc_conclusion
    }


    # Override Background
    override_background = {
        reference = council_chamber
    }


    left_portrait = {
        character = root
        animation = chancellor
    }


    right_portrait = {
        character = scope:power_holder

        # Animation if supporter is in entrenched regency. His regent is the one present
        triggered_animation = {
            trigger = {
                scope:main_supporter != scope:power_holder
            }
            animation = personality_rational
        }
        # Animation if supporter is BFF
        triggered_animation = {
            trigger = {
                root = { has_relation_best_friend = scope:main_supporter }
            }
            animation = ecstasy
        }
        # Animation if supporter is an ally
        triggered_animation = {
            trigger = {
                root = { is_allied_to = scope:main_supporter }
            }
            animation = interested
        }
        # Animation if supporter is being blackmailed
        triggered_animation = {
            trigger = {
                root = { has_strong_usable_hook = scope:main_supporter }
            }
            animation = shame
        }
    }
    
    lower_left_portrait = {
        character = scope:main_supporter
        trigger = {
            scope:main_supporter != scope:power_holder
        }
    }


    lower_center_portrait = {
        character = scope:helper
    }
    

    option = {
        name = drive_the_narrative.4.confirm
    }
    

    immediate = {
		play_music_cue = "mx_cue_faith_conversion"

        remove_trait = fugitive_of_the_plaguelands

        global_var:chosen_supporter = {
            save_scope_as = main_supporter
            # For the event background
            save_scope_as = bg_override_char

            if = {
                limit = {
                    has_diarchy_parameter = diarchy_type_is_entrenched_regency
                }

                diarch = { save_scope_as = power_holder }
            }
            else = {
                save_scope_as = power_holder
            }
        }





        # HELPER SELECTION
        # This is the guy that is going to help Calia during her speech

        # If the player has followers
        if = {
            limit = {
                any_courtier = {
                    is_available_ai_adult = yes
                }
            }


            # Witness
            if = {
                limit = {
                    employs_court_position = witness_camp_officer
                    any_court_position_holder = {
                        type = witness_camp_officer
                        is_available_ai_adult = yes
                    }
                }
                random_court_position_holder = {
                    type = witness_camp_officer
                    save_scope_as = helper
                }
            }
            # Laureate
            else_if = {
                limit = {
                    employs_court_position = laureate_camp_officer
                    any_court_position_holder = {
                        type = laureate_camp_officer
                        is_available_ai_adult = yes
                    }
                }
                random_court_position_holder = {
                    type = laureate_camp_officer
                    save_scope_as = helper
                }
            }
            # Second
            else_if = {
                limit = {
                    employs_court_position = second_camp_officer
                    any_court_position_holder = {
                        type = second_camp_officer
                        is_available_ai_adult = yes
                    }
                }
                random_court_position_holder = {
                    type = second_camp_officer
                    save_scope_as = helper
                }
            }
            # Check for someone appropriate in the player's camp
            else = {
                random_courtier = {
                    limit = {
                        is_available_ai_adult = yes
                    }

                    weight = {
                        base = 150
                        # Diplomat
                        modifier = {
				            has_trait = diplomat
                            add = 75
                        }
                        # # Gregarious
                        # modifier = {
				        #     has_trait = gregarious
                        #     add = 20
                        # }
                        # # Partner
                        # modifier = {
				        #     is_spouse_of = root
                        #     add = 10
                        # }
                        # # Family member
                        # modifier = {
				        #     is_close_family_of = root
                        #     add = 5
                        # }
                        # # High diplomacy
                        # modifier = { factor = diplomacy }
                    }

                    save_scope_as = helper
                }
            }
        }
        # If there are people at your supporter's court
        else_if = {
            limit = {
                scope:main_supporter = {
                    OR = {
                        any_courtier = {
                            is_available_ai_adult = yes
                        }

                        any_court_position_holder = {
                            is_available_ai_adult = yes
                        }
                    }
                }
            }


            # Chancellor
            scope:main_supporter = {
                if = {
                    limit = {
                        exists = cp:councillor_chancellor
                        cp:councillor_chancellor = { is_available_ai_adult = yes }
                    }
                    cp:councillor_chancellor = { save_scope_as = helper }
                }
                # Check for someone appropriate in the supporter's court
                else = {
                    random_courtier = {
                        limit = {
                            is_available_ai_adult = yes
                        }

                        weight = {
                            base = 1
                            # Diplomat
                            modifier = {
                                has_trait = diplomat
                                add = 30
                            }
                            # Gregarious
                            modifier = {
                                has_trait = gregarious
                                add = 20
                            }
                            # Partner
                            modifier = {
                                is_spouse_of = root
                                add = 10
                            }
                            # Family member
                            modifier = {
                                is_close_family_of = root
                                add = 5
                            }
                            # High diplomacy
                            modifier = { factor = diplomacy }
                        }
                        
                        save_scope_as = helper
                    }
                }
            }

        }
        # Fallback if there are noone at your or your supporter's court
        else = {
            # Create a helper
            create_character = {
                location = scope:main_supporter.capital_province
                template = tlm_diplomat_template
                save_scope_as = helper
                faith = scope:main_supporter.capital_province.faith
                culture = scope:main_supporter.capital_province.culture
            }
        }
    }
}
