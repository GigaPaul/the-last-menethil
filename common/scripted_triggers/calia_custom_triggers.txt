is_a_possible_supporter = {
    save_temporary_scope_as = target

    OR = {
        # The target is allied to the player
        is_allied_to = root

        # The player has a strong hook on the target
        root = { has_strong_usable_hook = scope:target }

        # The player is the target's BFF
        root = { has_relation_best_friend = scope:target }
    }
}

is_a_union_leader = {
    save_temporary_scope_as = target
    any_in_global_list = {
        variable = all_unions
        story_owner = target
    }
}

is_a_possible_supporter_and_leader = {
    # Target is the ruler of a union
    is_a_union_leader = yes
    is_a_possible_supporter = yes
}

tlm_knows_target = {
    OR = {
		has_relation_lover = $TARGET$
		has_relation_friend = $TARGET$
		has_relation_soulmate = $TARGET$
		has_relation_best_friend = $TARGET$
        is_close_family_of = $TARGET$
        is_consort_of = $TARGET$
        is_courtier_of = $TARGET$
        is_vassal_of = $TARGET$
    }
}
